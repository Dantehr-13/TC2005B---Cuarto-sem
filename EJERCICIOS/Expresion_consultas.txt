SELECT nombre, SUM(sueldo) AS ingresoTotal
FROM Elenco
GROUP BY nombre;

/*

*/
SELECT nomestudio, SUM(presupuesto) AS total
FROM Película
WHERE año BETWEEN 1980 AND 1989
GROUP BY nomestudio;

/*
3
*/
SELECT A.nombre, AVG(E.sueldo) AS sueldo_promedio
FROM Actor A
JOIN Elenco E ON A.nombre = E.nombre
WHERE A.sexo = 'M'
GROUP BY A.nombre
HAVING AVG(E.sueldo) > 5000000;

*/
SELECT titulo, año
FROM Película
WHERE presupuesto = (
    SELECT MIN(presupuesto)
    FROM Película
);

SELECT MAX(E.sueldo) AS mejor_sueldo
FROM Actor A
JOIN Elenco E ON A.nombre = E.nombre
WHERE A.sexo = 'F';
